Query-by-Committee with Kullback-Leibler Divergence¶

 9 import numpy as np
10 from matplotlib import pyplot as plt, animation
11 from sklearn.datasets import make_blobs
12 from skactiveml.utils import MISSING_LABEL, unlabeled_indices, labeled_indices
13 from skactiveml.visualization import plot_utility, plot_decision_boundary
14 from skactiveml.pool import QBC
15 from sklearn.ensemble import BaggingClassifier
16 from skactiveml.classifier import SklearnClassifier, PWC
17
18 random_state = np.random.RandomState(0)
19
20 # Build a dataset.
21 X, y_true = make_blobs(n_samples=10, n_features=2,
22                   centers=[[0, 1], [-3, .5], [-1, -1], [2, 1], [1, -.5]],
23                   cluster_std=.7, random_state=random_state)
24 y_true = y_true % 2
25 y = np.full(shape=y_true.shape, fill_value=MISSING_LABEL)
26
27 # Initialise the classifier.
28 clf = SklearnClassifier(BaggingClassifier(PWC()), classes=np.unique(y_true))
29 # Initialise the query strategy.
30 qs = QBC(method='KL_divergence', random_state=random_state)
31
32 # Preparation for plotting.
33 fig, ax = plt.subplots()
34 # bound = [[min(X[:, 0]), max(X[:, 0])], [min(X[:, 1]), max(X[:, 1])]]
35 feature_bound = [[min(X[:, 0]), min(X[:, 1])], [max(X[:, 0]), max(X[:, 1])]]
36 artists = []
37
38 # The active learning cycle:
39 n_cycles = 1
40 for c in range(n_cycles):
41     # Fit the classifier.
42     clf.fit(X, y)
43
44     # Set X_cand to the unlabeled instances.
45     unlbld_idx = unlabeled_indices(y)
46     X_cand = X[unlbld_idx]
47     X_labeled = X[labeled_indices(y)]
48
49     # Query the next instance/s.
50     query_params = {"ensemble": clf, "X": X, "y": y}
51     query_idx = unlbld_idx[qs.query(X_cand, **query_params)]
52
53     # Plot the labeled data.
54     coll_old = list(ax.collections)
55     title = ax.text(
56         0.5, 1.05, f"Decision boundry after acquring {c} labels",
57         size=plt.rcParams["axes.titlesize"], ha="center",
58         transform=ax.transAxes
59     )
60     ax = plot_utility(qs, {'ensemble': clf, 'X': X, 'y': y}, feature_bound=feature_bound, ax=ax)
61     ax.scatter(X[:, 0], X[:, 1], c=y_true, cmap="coolwarm", marker=".", zorder=2)
62     ax.scatter(X_labeled[:, 0], X_labeled[:, 1], c="grey", alpha=.8, marker=".", s=300)
63     ax = plot_decision_boundary(clf, feature_bound, ax=ax)
64
65     coll_new = list(ax.collections)
66     coll_new.append(title)
67     artists.append([x for x in coll_new if (x not in coll_old)])
68
69     # Label the queried instances.
70     y[query_idx] = y_true[query_idx]
71
72 ani = animation.ArtistAnimation(fig, artists, interval=1000, blit=True)

2
